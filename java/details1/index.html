# main 格式详细解释

    public class Demo_Static {
        public static void main(String[] args) {
        //  public: 被JVM调用, 所以权限要足够大
        //  static: 被JVM调用, 不需要创建对象, 直接类名.调用即可
        //  void: 被JVM调用, 不需要任何返回值
        //  main: 只有这样写才能被jvm识别 mian 不是关键字
        //  String[] args: 接收键盘录入
        }
    }
